1function [rrc] = rrcpulse(t,rolloff,T,OS)
 2
 3% root raised cosine pulse
 4%
 5% INPUT:
 6%   t       - vector of time instants at which pulse is evaluated
 7%   rolloff - rolloff factor
 8%   T       - symbol period
 9%   OS      - oversampling factor
10
11Ts = T/OS;    % sample period
12fc = 1/(2*T); % cut off frequency
13fs = 1/Ts;    % sampling frequency
14
15% implementation based on MATLAB's firrcos
16
17% use logical indexing:
18tol = 2*eps;
19idx1 = (abs(t) < tol);
20% idx2 = (abs(abs(8*rolloff*fc*t) - 1.0) < sqrt(eps));
21idx2 = (abs(abs(8*rolloff*fc*t) - 1.0) < tol);
22idx3 = ~idx1 & ~idx2;
23
24rrc = zeros(size(t));
25rrc(idx1) = -sqrt(2.*fc) ./ (pi.*fs) .* (pi.*(rolloff-1) - 4.*rolloff );
26rrc(idx2) = sqrt(2.*fc) ./ (2.*pi.*fs) ...
27        * (    pi.*(rolloff+1)  .* sin(pi.*(rolloff+1)./(4.*rolloff)) ...
28        - 4.*rolloff .* sin(pi.*(rolloff-1)./(4.*rolloff)) ...
29        + pi.*(rolloff-1) .* cos(pi.*(rolloff-1)./(4.*rolloff)) ...
30        );
31rrc(idx3) = -4.*rolloff./fs .* ( cos((1+rolloff).*2.*pi.*fc.*t(idx3)) + ...
32        sin((1-rolloff).*2.*pi.*fc.*t(idx3)) ./ (8.*rolloff.*fc.*t(idx3)) ) ...
33        ./ (pi .* sqrt(1./(2.*fc)) .* ((8.*rolloff.*fc.*t(idx3)).^2 - 1));
34    
35% scale pulse
36rrc = rrc * sqrt(2.*fc);
37
38end
