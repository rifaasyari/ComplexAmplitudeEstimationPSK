1function c  = conv_fft(a, b)
 2%compute a convolution using the fft.  The output is identical to:
 3%conv(a,b,`valid')  when the length(a) >= length(b),  and
 4%conv(b,a,`valid') when length(a) =< length(b).
 5%
 6% (c) 2012 ITR-UniSA
 7% Authors: Robby McKilliam
 8% Created: 3 April 2012
 9    
10    P = numel(a);
11    Q = numel(b);
12    L = P + Q - 1;
13    M = min([P,Q]);
14
15    c = ifft(fft(a,L) .* fft(b,L));
16    c = c(M:end-M+1);
17
18end
