  539       }
  540   
  541       emxFree_real_T(&costab1q);
  542       ix = 0;
  543       chanStart = 0;
  544       while ((nRows > 0) && (chanStart <= lastChan)) {
  545         ju = 0U;
  546         nd2 = chanStart;
  547         for (i = 1; i <= nRowsM1; i++) {
  548           y->data[nd2] = x->data[ix];
  549           c_n = n;
  550           tst = TRUE;
  551           while (tst) {
  552             c_n >>= 1U;
  553             ju ^= c_n;
  554             tst = ((int32_T)(ju & c_n) == 0);
  555           }
  556   
  557           nd2 = chanStart + (int32_T)ju;
  558           ix++;
  559         }
  560   
  561         y->data[nd2] = x->data[ix];
  562         ix += ixDelta;
  563         nd2 = (chanStart + nRows) - 2;
  564         if (nRows > 1) {
  565           for (i = chanStart; i <= nd2; i += 2) {
  566             temp_re = y->data[i + 1].re;
  567             temp_im = y->data[i + 1].im;
  568             y->data[i + 1].re = y->data[i].re - y->data[i + 1].re;
  569             y->data[i + 1].im = y->data[i].im - y->data[i + 1].im;
  570             y->data[i].re += temp_re;
  571             y->data[i].im += temp_im;
  572           }
  573         }
  574   
  575         n2 = 2;
  576         iDelta2 = 4;
  577         k = nRowsD4;
  578         iheight = 1 + ((nRowsD4 - 1) << 2);
  579         while (k > 0) {
  580           i = chanStart;
  581           ihi = chanStart + iheight;
  582           while (i < ihi) {
  583             nd2 = i + n2;
  584             temp_re = y->data[nd2].re;
  585             temp_im = y->data[nd2].im;
  586             y->data[i + n2].re = y->data[i].re - y->data[nd2].re;
  587             y->data[i + n2].im = y->data[i].im - y->data[nd2].im;
  588             y->data[i].re += temp_re;
  589             y->data[i].im += temp_im;
  590             i += iDelta2;
  591           }
  592   
  593           nd2 = chanStart + 1;
  594           for (b_n = k; b_n < nRowsD2; b_n += k) {
  595             e = costab->data[b_n];
  596             twid_im = sintab->data[b_n];
  597             i = nd2;
  598             ihi = nd2 + iheight;
  599             while (i < ihi) {
  600               temp_re = e * y->data[i + n2].re - twid_im * y->data[i + n2].im;
  601               temp_im = e * y->data[i + n2].im + twid_im * y->data[i + n2].re;
  602               y->data[i + n2].re = y->data[i].re - temp_re;
  603               y->data[i + n2].im = y->data[i].im - temp_im;
  604               y->data[i].re += temp_re;
  605               y->data[i].im += temp_im;
  606               i += iDelta2;
  607             }
  608   
  609             nd2++;
  610           }
  611   
  612           k /= 2;
  613           n2 = iDelta2;
  614           iDelta2 <<= 1;
  615           iheight -= n2;
  616         }
  617   
  618         chanStart += nRows;
  619       }
  620   
  621       emxFree_real_T(&sintab);
  622       emxFree_real_T(&costab);
  623       if (y->size[0] > 1) {
  624         e = 1.0 / (real_T)y->size[0];
  625         iDelta2 = y->size[0];
  626         emxEnsureCapacity((emxArray__common *)y, iDelta2, (int32_T)sizeof(creal_T));
  627         nd2 = y->size[0] - 1;
  628         for (iDelta2 = 0; iDelta2 <= nd2; iDelta2++) {
  629           y->data[iDelta2].re *= e;
  630           y->data[iDelta2].im *= e;
  631         }
  632       }
  633     }
  634   }

  681     boolean_T exitg2;

  711     emxArray_real_T *b_lvec;

  714     emxArray_creal_T *b_y1;

  722     /* Run a matched filter on the received samples, returns recieved */
  723     /* symbols. */
  724     /*      */
  725     /*     mfout = mf(rx, tauhat, acqparams) */
  726     /*  */
  727     /*  INPUTS: */
  728     /*    rx              - sampled received signal */
  729     /*    tauhat        - time offset estimate */
  730     /*    acqparam        - struct holding aquisition parameters */
  731     /*        .T              - symbol period */
  732     /*        .L              - number of symbols (assumed consequtive) */
  733     /*        .OS             - oversampling factor */
  734     /*        .pshape         - function representing transmission pulse shape */
  735     /*  */
  736     /*  OUTPUTS: */
  737     /*    mfout        - matched filtered signal at symbol rate */
  738     /*  (c) 2012 ITR-UniSA */
  739     /*  Authors: Robby McKilliam */
  740     /*  Created: 4 April 2012 */
  741     /* g = acqparam.pshape; */
  742     /* 'mf_coder:24' p = 1; */
  743     /* q = acqparam.OS; */
  744     /* T = acqparam.T; */
  745     /* 'mf_coder:27' Ts = p*T/q; */
  746     Ts = mrdivide(T, q);
  747   
  748     /* 'mf_coder:28' mini = min(min(P),min(D)); */
  749     ixstart = 1;
  750     mtmp = P[0];
  751     if (rtIsNaN(P[0])) {
  752       nm1d2 = 2;
  753       exitg4 = 0U;
  754       while ((exitg4 == 0U) && (nm1d2 < 31)) {
  755         ixstart = nm1d2;
  756         if (!rtIsNaN(P[nm1d2 - 1])) {
  757           mtmp = P[nm1d2 - 1];
  758           exitg4 = 1U;
  759         } else {
  760           nm1d2++;
  761         }
  762       }
  763     }
  764   
  765     if (ixstart < 30) {
  766       while (ixstart + 1 < 31) {
  767         if (P[ixstart] < mtmp) {
  768           mtmp = P[ixstart];
  769         }
  770   
  771         ixstart++;
  772       }
  773     }
  774   
  775     ixstart = 1;
  776     ndbl = D[0];
  777     if (rtIsNaN(D[0])) {
  778       nm1d2 = 2;
  779       exitg3 = 0U;
  780       while ((exitg3 == 0U) && (nm1d2 < 3961)) {
  781         ixstart = nm1d2;
  782         if (!rtIsNaN(D[nm1d2 - 1])) {
  783           ndbl = D[nm1d2 - 1];
  784           exitg3 = 1U;
  785         } else {
  786           nm1d2++;
  787         }
  788       }
  789     }
  790   
  791     if (ixstart < 3960) {
  792       while (ixstart + 1 < 3961) {
  793         if (D[ixstart] < ndbl) {
  794           ndbl = D[ixstart];
  795         }
  796   
  797         ixstart++;
  798       }
  799     }
  800   
  801     mini = (mtmp <= ndbl) || rtIsNaN(ndbl) ? mtmp : ndbl;
  802   
  803     /* 'mf_coder:29' maxi = max(max(P),max(D)); */
  804     ixstart = 1;
  805     mtmp = P[0];
  806     if (rtIsNaN(P[0])) {
  807       nm1d2 = 2;
  808       exitg2 = 0U;
  809       while ((exitg2 == 0U) && (nm1d2 < 31)) {
  810         ixstart = nm1d2;
  811         if (!rtIsNaN(P[nm1d2 - 1])) {
  812           mtmp = P[nm1d2 - 1];
  813           exitg2 = 1U;
  814         } else {
  815           nm1d2++;
  816         }
  817       }
  818     }
  819   
  820     if (ixstart < 30) {
  821       while (ixstart + 1 < 31) {
  822         if (P[ixstart] > mtmp) {
  823           mtmp = P[ixstart];
  824         }
  825   
  826         ixstart++;
  827       }
  828     }
  829   
  830     ixstart = 1;
  831     ndbl = D[0];
  832     if (rtIsNaN(D[0])) {
  833       nm1d2 = 2;
  834       exitg1 = 0U;
  835       while ((exitg1 == 0U) && (nm1d2 < 3961)) {
  836         ixstart = nm1d2;
  837         if (!rtIsNaN(D[nm1d2 - 1])) {
  838           ndbl = D[nm1d2 - 1];
  839           exitg1 = 1U;
  840         } else {
  841           nm1d2++;
  842         }
  843       }
  844     }
  845   
  846     if (ixstart < 3960) {
  847       while (ixstart + 1 < 3961) {
  848         if (D[ixstart] > ndbl) {
  849           ndbl = D[ixstart];
  850         }
  851   
  852         ixstart++;
  853       }
  854     }
  855   
  856     maxi = (mtmp >= ndbl) || rtIsNaN(ndbl) ? mtmp : ndbl;
  857   
  858     /* 'mf_coder:30' N = length(rx); */
  859     /* 'mf_coder:31' L = maxi; */
  860     /* 'mf_coder:33' ns = (p*q*mini - p*N : p*q*maxi - p)*(-1)*Ts/p - tauhat; */
  861     y = q * mini;
  862     b_y = q * maxi;
  863     mtmp = q * mini - 16000.0;
  864     ndbl = q * maxi - 1.0;
  865     if (rtIsNaN(mtmp) || rtIsNaN(ndbl)) {
  866       n = 1;
  867       mtmp = rtNaN;
  868       apnd = b_y - 1.0;
  869     } else if (b_y - 1.0 < y - 16000.0) {
  870       n = 0;
  871       mtmp = y - 16000.0;
  872       apnd = b_y - 1.0;
  873     } else if (rtIsInf(mtmp) || rtIsInf(ndbl)) {
  874       n = 1;
  875       mtmp = rtNaN;
  876       apnd = b_y - 1.0;
  877     } else {
  878       mtmp = y - 16000.0;
  879       ndbl = floor(((b_y - 1.0) - (y - 16000.0)) + 0.5);
  880       apnd = (y - 16000.0) + ndbl;
  881       cdiff = apnd - (b_y - 1.0);
  882       absa = fabs(y - 16000.0);
  883       absb = fabs(b_y - 1.0);
  884       if (absa > absb) {
  885         absb = absa;
  886       }
  887   
  888       if (fabs(cdiff) < 4.4408920985006262E-16 * absb) {
  889         ndbl++;
  890         apnd = b_y - 1.0;
  891       } else if (cdiff > 0.0) {
  892         apnd = (y - 16000.0) + (ndbl - 1.0);
  893       } else {
  894         ndbl++;
  895       }
  896   
  897       if (ndbl >= 0.0) {
  898         n = (int32_T)ndbl;
  899       } else {
  900         n = 0;
  901       }
  902     }
  903   
  904     emxInit_real_T(&ns, 2);
  905     i7 = ns->size[0] * ns->size[1];
  906     ns->size[0] = 1;
  907     ns->size[1] = n;
  908     emxEnsureCapacity((emxArray__common *)ns, i7, (int32_T)sizeof(real_T));
  909     if (n > 0) {
  910       ns->data[0] = mtmp;
  911       if (n > 1) {
  912         ns->data[n - 1] = apnd;
  913         ixstart = n - 1;
  914         nm1d2 = ixstart / 2;
  915         for (k = 1; k <= nm1d2 - 1; k++) {
  916           ns->data[k] = mtmp + (real_T)k;
  917           ns->data[(n - k) - 1] = apnd - (real_T)k;
  918         }
  919   
  920         if (nm1d2 << 1 == ixstart) {
  921           ns->data[nm1d2] = (mtmp + apnd) / 2.0;
  922         } else {
  923           ns->data[nm1d2] = mtmp + (real_T)nm1d2;
  924           ns->data[nm1d2 + 1] = apnd - (real_T)nm1d2;
  925         }
  926       }
  927     }
  928   
  929     i7 = ns->size[0] * ns->size[1];
  930     ns->size[0] = 1;
  931     emxEnsureCapacity((emxArray__common *)ns, i7, (int32_T)sizeof(real_T));
  932     ixstart = ns->size[0];
  933     nm1d2 = ns->size[1];
  934     nm1d2 = ixstart * nm1d2 - 1;
  935     for (i7 = 0; i7 <= nm1d2; i7++) {
  936       ns->data[i7] = -ns->data[i7] * Ts - tauhat;
  937     }
  938   
  939     emxInit_boolean_T(&idx1, 2);
  940     emxInit_real_T(&r10, 2);
  941   
  942     /* lvec = conj( g(-ns*Ts/p - tauhat) ); */
  943     /* 'mf_coder:35' lvec = conj( rrcpulse(ns,rolloff,T,q) .*... */
  944     /* 'mf_coder:36'     (abs(ns) <= RrcNsym*q*Ts)); */
  945     /*  root raised cosine pulse */
  946     /*  */
  947     /*  INPUT: */
  948     /*    t       - vector of time instants at which pulse is evaluated */
  949     /*    rolloff - rolloff factor */
  950     /*    T       - symbol period */
  951     /*    OS      - oversampling factor */
  952     /* 'rrcpulse:11' Ts = T/OS; */
  953     /*  sample period */
  954     /* 'rrcpulse:12' fc = 1/(2*T); */
  955     apnd = 1.0 / (2.0 * T);
  956   
  957     /*  cut off frequency */
  958     /* 'rrcpulse:13' fs = 1/Ts; */
  959     cdiff = 1.0 / (T / q);
  960   
  961     /*  sampling frequency */
  962     /*  implementation based on MATLAB's firrcos */
  963     /*  use logical indexing: */
  964     /* 'rrcpulse:18' tol = 2*eps; */
  965     /* 'rrcpulse:19' idx1 = (abs(t) < tol); */
  966     c_abs(ns, r10);
  967     i7 = idx1->size[0] * idx1->size[1];
  968     idx1->size[0] = 1;
  969     idx1->size[1] = r10->size[1];
  970     emxEnsureCapacity((emxArray__common *)idx1, i7, (int32_T)sizeof(boolean_T));
  971     nm1d2 = r10->size[0] * r10->size[1] - 1;
  972     for (i7 = 0; i7 <= nm1d2; i7++) {
  973       idx1->data[i7] = (r10->data[i7] < 4.4408920985006262E-16);
  974     }
  975   
  976     emxInit_real_T(&c_y, 2);
  977   
  978     /*  idx2 = (abs(abs(8*rolloff*fc*t) - 1.0) < sqrt(eps)); */
  979     /* 'rrcpulse:21' idx2 = (abs(abs(8*rolloff*fc*t) - 1.0) < tol); */
  980     y = 8.0 * rolloff * apnd;
  981     i7 = c_y->size[0] * c_y->size[1];
  982     c_y->size[0] = 1;
  983     c_y->size[1] = ns->size[1];
  984     emxEnsureCapacity((emxArray__common *)c_y, i7, (int32_T)sizeof(real_T));
  985     nm1d2 = ns->size[0] * ns->size[1] - 1;
  986     for (i7 = 0; i7 <= nm1d2; i7++) {
  987       c_y->data[i7] = y * ns->data[i7];
  988     }
  989   
  990     emxInit_real_T(&r11, 2);
  991     c_abs(c_y, r10);
  992     i7 = r11->size[0] * r11->size[1];
  993     r11->size[0] = 1;
  994     r11->size[1] = r10->size[1];
  995     emxEnsureCapacity((emxArray__common *)r11, i7, (int32_T)sizeof(real_T));
  996     emxFree_real_T(&c_y);
  997     nm1d2 = r10->size[0] * r10->size[1] - 1;
  998     for (i7 = 0; i7 <= nm1d2; i7++) {
  999       r11->data[i7] = r10->data[i7] - 1.0;
 1000     }
 1001   
 1002     emxInit_boolean_T(&idx2, 2);
 1003     c_abs(r11, r10);
 1004     i7 = idx2->size[0] * idx2->size[1];
 1005     idx2->size[0] = 1;
 1006     idx2->size[1] = r10->size[1];
 1007     emxEnsureCapacity((emxArray__common *)idx2, i7, (int32_T)sizeof(boolean_T));
 1008     emxFree_real_T(&r11);
 1009     nm1d2 = r10->size[0] * r10->size[1] - 1;
 1010     for (i7 = 0; i7 <= nm1d2; i7++) {
 1011       idx2->data[i7] = (r10->data[i7] < 4.4408920985006262E-16);
 1012     }
 1013   
 1014     emxInit_boolean_T(&idx3, 2);
 1015   
 1016     /* 'rrcpulse:22' idx3 = ~idx1 & ~idx2; */
 1017     i7 = idx3->size[0] * idx3->size[1];
 1018     idx3->size[0] = 1;
 1019     idx3->size[1] = idx1->size[1];
 1020     emxEnsureCapacity((emxArray__common *)idx3, i7, (int32_T)sizeof(boolean_T));
 1021     nm1d2 = idx1->size[0] * idx1->size[1] - 1;
 1022     for (i7 = 0; i7 <= nm1d2; i7++) {
 1023       idx3->data[i7] = ((!idx1->data[i7]) && (!idx2->data[i7]));
 1024     }
 1025   
 1026     /* 'rrcpulse:24' rrc = zeros(size(t)); */
 1027     for (i7 = 0; i7 < 2; i7++) {
 1028       varargin_1[i7] = (uint32_T)ns->size[i7];
 1029     }
 1030   
 1031     emxInit_real_T(&lvec, 2);
 1032     i7 = lvec->size[0] * lvec->size[1];
 1033     lvec->size[0] = 1;
 1034     emxEnsureCapacity((emxArray__common *)lvec, i7, (int32_T)sizeof(real_T));
 1035     i7 = lvec->size[0] * lvec->size[1];
 1036     lvec->size[1] = (int32_T)varargin_1[1];
 1037     emxEnsureCapacity((emxArray__common *)lvec, i7, (int32_T)sizeof(real_T));
 1038     nm1d2 = (int32_T)varargin_1[1] - 1;
 1039     for (i7 = 0; i7 <= nm1d2; i7++) {
 1040       lvec->data[i7] = 0.0;
 1041     }
 1042   
 1043     b_emxInit_int32_T(&r12, 2);
 1044     b_emxInit_int32_T(&r13, 2);
 1045   
 1046     /* 'rrcpulse:25' rrc(idx1) = -sqrt(2.*fc) ./ (pi.*fs) .* (pi.*(rolloff-1) - 4.*rolloff ); */
 1047     eml_li_find(idx1, r13);
 1048     i7 = r12->size[0] * r12->size[1];
 1049     r12->size[0] = 1;
 1050     r12->size[1] = r13->size[1];
 1051     emxEnsureCapacity((emxArray__common *)r12, i7, (int32_T)sizeof(int32_T));
 1052     nm1d2 = r13->size[0] * r13->size[1] - 1;
 1053     for (i7 = 0; i7 <= nm1d2; i7++) {
 1054       r12->data[i7] = r13->data[i7];
 1055     }
 1056   
 1057     mtmp = -sqrt(2.0 * apnd) / (3.1415926535897931 * cdiff);
 1058     nm1d2 = r12->size[0] * r12->size[1] - 1;
 1059     for (i7 = 0; i7 <= nm1d2; i7++) {
 1060       lvec->data[r12->data[i7] - 1] = mtmp * (3.1415926535897931 * (rolloff - 1.0)
 1061         - 4.0 * rolloff);
 1062     }
