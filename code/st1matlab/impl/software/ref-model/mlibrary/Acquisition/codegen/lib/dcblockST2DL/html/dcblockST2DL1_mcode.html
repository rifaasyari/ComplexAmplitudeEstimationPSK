1% DC blocking filter
 2%-------------------------------------------------------------
 3% dcblock.m 
 4%
 5% [RxSampFilt] = dcblockST2DL(RxSamp)
 6%
 7% This digital filter blocks the DC component in the received signal. It is
 8% primarily intended to remove the complex exponential at DC that is
 9% transmitted in the ST2 downlink waveform for frequency estimation
10% purposes. Once the frequency has been estimated, the discrete tone should
11% be removed in order to avoid bit error rate degradations. The
12% implementation uses an order-one IIR filter.
13% 
14% Inputs:
15%   RxSamp      - Vector containing samples of received signal
16%
17% Outputs:
18%   RxSampFilt  - Vector containing filtered received samples
19%
20% Comments: For details see [J. de Freitas, "The DC Blocking Filter", Jan.
21%           2007, available at
22%           http://www.mathworks.com.au/matlabcentral/fileexchange/13792-the-dc-blocking-filter] 
23% 
24%-------------------------------------------------------------
25% Author: A. Pollok, Institute for Telecommunications Research
26% Project: ASRP
27%-------------------------------------------------------------
28% Copyright 2012 
29% Institute for Telecommunications Research
30% University of South Australia
31%-------------------------------------------------------------
32
33function [RxSampFilt] = dcblockST2DL(RxSamp)
34
35% ST1 UL: coefficents of order-one IIR filter
36
37p = 0.998009283733233; % coeff for cut-on frequency at 12Hz
38    % fc = 12 / (OS/(2*Ts)); % normalised cut-on frequency
39    % p = dcblock_MatlabCentral(fc);
40b = [1 -1];
41a = [1 -p];
42
43% filter frequency corrected samples
44RxSampFilt = filter(b,a,RxSamp);
45
46end
