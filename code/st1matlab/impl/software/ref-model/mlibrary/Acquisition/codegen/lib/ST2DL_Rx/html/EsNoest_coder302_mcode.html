1% Joint pilot and data based Es/No estimator.
 2%-------------------------------------------------------------
 3% EsNoest.m 
 4%
 5% [EsNohat] = EsNoest(MFoutput,AcqParams)
 6%
 7% This function etimates the Es/No from the complex matched filter (MF)
 8% output. First, hard decisions for the data symbols are obtained from the
 9% MF output. After inserting the known pilot symbols, the symbol stream is
10% used to strip the modulation of the MF output. The Es/No is estimated
11% from the resulting signal.
12% 
13% Inputs:
14%   MFoutput    - Vector containing complex matched filter output at one 
15%                 sample per symbol.
16%   AcqParams   - Struct holding aquisition parameters
17%       .P              - position (indices) of pilot symbols
18%       .p              - pilot symbols
19%       .D              - position (indices) of data symbols
20%       .M              - M for M-PSK, i.e. size of constellation
21%
22% Outputs:
23%   EsNohat     - Estimate of the Es/No.
24%
25% Comments:
26% 
27%-------------------------------------------------------------
28% Author: A. Pollok, Institute for Telecommunications Research
29% Project: ASRP
30%-------------------------------------------------------------
31% Copyright 2012 
32% Institute for Telecommunications Research
33% University of South Australia
34%-------------------------------------------------------------
35
36function [EsNohat] = EsNoest_coder(MFoutput,M,D,P,p)
37
38symbhat = zeros(1,length(P)+length(D))+1i*zeros(1,length(P)+length(D));
39% make hard decisions for PSK data symbols
40symbsectorhat = mod(round(angle(MFoutput(D))/(2*pi/M)), M);
41symbhat(D) = exp(2i*pi/M*symbsectorhat);
42% insert pilot symbols
43symbhat(P) = p;
44
45% Bill's blind Es/No estimator
46MFoutDemod = MFoutput .* conj(symbhat); % strip modulation off
47EsNohat = abs(mean(MFoutDemod) / std(MFoutDemod))^2;
48
49end
